
clc;
clear all;
close all;


n = 3;

taps = [3 2];

reg = [1 0 0]; 

seq_len = (2^n) - 1;

pn_seq = zeros(1, seq_len);

for i = 1:seq_len
    pn_seq(i) = reg(end); % Output bit (rightmost bit)
    
    feedback = 0;
    for t = 1:length(taps)
        feedback = xor(feedback, reg(n - taps(t) + 1));
    end
    
   
    reg = [feedback reg(1:end-1)];
end

disp('Generated PN Sequence: ');
disp(pn_seq);
